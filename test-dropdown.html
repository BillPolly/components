Testing Server File Loading

Open the developer console and run these commands to test the dropdown:

// 1. Select the dropdown
const dropdown = document.querySelector('[data-demo="basic"] select[data-action="load-server-file"]');

// 2. Load package.json
dropdown.value = '/data/samples/package.json';
dropdown.dispatchEvent(new Event('change', { bubbles: true }));

// 3. Check status after a moment
setTimeout(() => {
    const status = document.querySelector('[data-status="basic"]');
    console.log('Status:', status.textContent);
}, 1000);

// 4. Try other files
dropdown.value = '/data/samples/user-profile.xml';
dropdown.dispatchEvent(new Event('change', { bubbles: true }));
