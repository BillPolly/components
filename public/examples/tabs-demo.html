Usage Example

// Create tabs with MVVM architecture const tabs = Tabs.create({ dom: document.getElementById('my-tabs'), tabs: [ { id: 'home', title: 'Home', content: '<h3>Welcome</h3><p>Welcome to our app!</p>', icon: 'üè†' }, { id: 'profile', title: 'Profile', content: '<h3>User Profile</h3><p>Manage your profile here.</p>', badge: '3' }, { id: 'settings', title: 'Settings', content: '<h3>Settings</h3><p>Configure your preferences.</p>', closable: true } ], // Theme and appearance theme: 'light', variant: 'pills', position: 'top', // Behavior allowClosable: true, allowReorderable: true, keyboardNavigation: true, showAddButton: true, // Event callbacks onActiveTabChange: (tabId, tab, previousTabId) => { console.log('Active tab changed:', tabId); }, onTabAdd: (tabId, tab) => { console.log('Tab added:', tab.title); }, onTabRemove: (tabId, tab) => { console.log('Tab removed:', tab.title); }, onAdd: () => { // Custom add button behavior const newTabId = tabs.addTab({ title: `Tab ${tabs.getTabs().length + 1}`, content: `<p>Content for new tab</p>`, closable: true }); tabs.setActiveTab(newTabId); } }); // Programmatic control tabs.addTab({ title: 'Dynamic Tab', icon: '‚ö°' }); tabs.setTabBadge('profile', '5'); tabs.updateTab('settings', { title: 'New Settings', icon: '‚öôÔ∏è' }); tabs.setTheme('dark'); // State management const state = tabs.exportState(); localStorage.setItem('tabsState', JSON.stringify(state));
